function readCodexSortState(config) {
  const sortMode =
    document.getElementById(config.sortId)?.value ||
    config.selectedSort;

  const direction =
    document.getElementById(config.directionId)?.dataset?.direction ||
    "asc";

  return {
    sortMode,
    direction
  };
}
